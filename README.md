# mathtool
#BUILD
make  
sudo make install  
#RUN
htod : convert HEX to DEC  
       Ex, htod fee1c001  
  
dtoh : convert DEC to HEX  
       Ex, dtoh 4276207617  

gettimeofday : get current time(UNIX timestamp)
       Ex, gettimeofday
  
ctime : convert UNIX timestamp to string
       Ex, ctime 1461570588 1461570598
